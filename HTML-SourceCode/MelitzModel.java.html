103      public void buildModel() { 
104          System.out.println("Building model"); 
105          schedule = new Schedule(1); 
106          //Trade Costs are symmetric 
107          //Containers for global economic variables 
108   
109          //we assume two regions, pg. 6: More regions can be added here 
110          countryEconomyList = new ArrayList<Economy>(); 
111   
112          //North this the baseline 
113          NorthEconomy = new Economy(SIM_TIME, KAPITAL_SHARE * SOUTH_KAPITAL_ENDOWMENT, LABOR_SHARE * SOUTH_LABOR_ENDOWMENT, "North", TRADE_COSTS, countryEconomyList, 1, 1, 1, 1); 
114          //South has values with respect to north 
115          SouthEconomy = new Economy(SIM_TIME, SOUTH_KAPITAL_ENDOWMENT, SOUTH_LABOR_ENDOWMENT, "South", TRADE_COSTS, countryEconomyList, OVERHEAD_DIFFERENTIAL, DEPRECIATION_DIFFERENTIAL, BARRIERS_ENTRY_DIFFERENTIAL, 2); 
116   
117          //Order matters !! Don't shuffle country lists 
118          countryEconomyList.add(NorthEconomy);  // index =1 for easier String-free manipulations, and extensibility 
119          countryEconomyList.add(SouthEconomy);  // index =2 
120   
121          //World Economy Initalization, world government gets a bunch of 0s for institutional parameters because after all there is no world government...this avoids prgramming errors by throwing bizarre 
122          //values if called 
123          WorldEconomy = new Economy(SIM_TIME, (NorthEconomy.manufacturers.size() + SouthEconomy.manufacturers.size()), (NorthEconomy.Labor + SouthEconomy.Labor), "World", TRADE_COSTS, countryEconomyList, 0, 0, 0, 0); 
124          countryEconomyList.add(0, WorldEconomy);  //0 is all the economies. 
125      }

143      public void buildSchedule() { 
144          System.out.println("Building schedule"); 
145          class TradeModelDailyStep extends BasicAction { 
146              public void execute() { 
147   
148                  //Flag for GUI to boost performance for batch calculations 
149                  if (GUI) { 
150                      guiUpdate(); 
151                  } 
152   
153                  //System.out.println("Northern productivity " + Economy.getAverageProductivity(NorthEconomy)); 
154                  //System.out.println("Southern productivity " + Economy.getAverageProductivity(SouthEconomy)); 
155   
156                  //Relocation algorithm 
157                  if (BALDWINIAN_RELOCATION) { 
158                      System.out.println("I might be missing out on profit, let's see if I should relocate today..."); 
159                      int bestProducer = Firm.findMostProductive(SouthEconomy.manufacturers); 
160                      System.out.println("My productivity is = " + SouthEconomy.manufacturers.get(bestProducer).productivity); 
161                      double currentLocationProfit = Firm.profit(SouthEconomy.manufacturers.get(bestProducer), countryEconomyList, 2, (int) schedule.getCurrentTime()); 
162                      double expectedRelocatedProfit = Firm.profit(SouthEconomy.manufacturers.get(bestProducer), countryEconomyList, 1, (int) schedule.getCurrentTime()); 
163   
164                      System.out.println("If I stay, my profit is " + currentLocationProfit + ". If I go, my profit is " + expectedRelocatedProfit + ". The difference= " + (expectedRelocatedProfit - currentLocationProfit)); 
165                      if (expectedRelocatedProfit > currentLocationProfit) { 
166                          System.out.println("OK, let's move then..."); 
167                          Firm.relocate(countryEconomyList, 2, 1, bestProducer);  //From South (2) to North (1) 
168                      } 
169                  } 
170                  //Based on Hopenhayn 1992 1130: #A4): Depreciates a part of the capital stock, then adds new entrants who receive a based productivity draw & competition in local market 
171                  if (ENDOGENOUS_GROWTH) { 
172   
173                      double mean; 
174                      double variance; 
175                      double prodGrowth; 
176   
177                      for (int i = 1; i < countryEconomyList.size(); i++) { 
178                          double ZeroProfitFirmProductivity = Firm.findLeastProductive(countryEconomyList.get(i).manufacturers); 
179                          mean = Firm.findAverageProductivity(countryEconomyList, i); 
180                          variance = Math.sqrt(mean); 
181                          prodGrowth = mean * (.001); 
182                          System.out.println("Country = " + i + "; mean productivity=" + mean + "; shock variance= " + variance + "; Daily productivity growth= " + prodGrowth); 
183                          for (int k = 0; k < countryEconomyList.get(k).manufacturers.size(); k++) { 
184                              //Apply Hopenhayn type shock 
185                              countryEconomyList.get(i).manufacturers.get(k).productivity = countryEconomyList.get(i).manufacturers.get(k).productivity + Random.normal.nextDouble((mean - countryEconomyList.get(i).government.depreciation), variance); 
186                              //In equilibrium firms exit the industry whenever their state falls below a reservation level x. 
187                              if (countryEconomyList.get(i).manufacturers.get(k).productivity < ZeroProfitFirmProductivity) { 
188                                  countryEconomyList.get(i).manufacturers.remove(k); 
189                                  //Location and shape parameters for the productivity draw based on country economy, plus competition adjustment 
190                                  countryEconomyList.get(i).manufacturers.add(k, new Firm(i, (countryEconomyList.get(i).technologicalFrontier + prodGrowth), countryEconomyList.get(i).technologicalFrontier, countryEconomyList.get(i))); 
191                              } 
192                          } 
193                      } 
194                  } 
195   
196                  System.out.println("Average World prod = " + Firm.findAverageProductivity(countryEconomyList, 0) + "; Northern prod = " + Firm.findAverageProductivity(countryEconomyList, 1) + "; Southern prod = " + Firm.findAverageProductivity(countryEconomyList, 2)); 
197                  //Collecting consumer income 
198                  Consumer.collectIncome(countryEconomyList, (int) schedule.getCurrentTime()); 
199                  //Updating national price levels to reflect market competition after migration 
200                  for (int i = 0; i < countryEconomyList.size(); i++) { 
201                      countryEconomyList.get(i).aggregatePrice = Economy.getAveragePrice(countryEconomyList.get(i).manufacturers); 
202                      System.out.println("Country=" + i + "; Price = " + countryEconomyList.get(i).aggregatePrice); 
203                  } 
204              } 
205   
206          } 
207          schedule.scheduleActionAtInterval(1, new TradeModelDailyStep()); 
208          schedule.scheduleActionAtEnd(this, "outPut"); 
209      }
